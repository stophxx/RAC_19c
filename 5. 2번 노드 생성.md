```
# hostnamectl set-hostname jjh2
```


-**oracle 계정의 bash_profile을 변경한다.**

```
[root@jjh2 ~]# vi ~oracle/.bash_profile

export ORACLE_BASE=/u01/app/oracle;
export ORACLE_HOME=$ORACLE_BASE/product/19c;
export ORACLE_SID=jjh2;        #2번노드 jjh2
export ORACLE_HOSTNAME=jjh2;   #2번노드 jjh2
export ORACLE_UNQNAME=jjhdb;
export GRID_HOME=/u01/app/19c/grid;
export GRID_SID=+ASM2;         #2번노드 ASM2
export PATH=$ORACLE_HOME/bin:$GRID_HOME/bin:$PATH
export LD_LIBRARY_PATH=$ORACLE_HOME/lib:/lib:/usr/lib;
export CLASSPATH=$ORACLE_HOME/jlib:$ORACLE_HOME/rdbms/jlib;
export EDITOR=vi;

alias grid='export ORACLE_HOME=$GRID_HOME; export ORACLE_SID=$GRID_SID; export PATH=$ORACLE_HOME/bin:$GRID_HOME/bin:$PATH; echo $ORACLE_SID; echo $ORACLE_HOME'
alias db='. ~oracle/.bash_profile;export PATH=$ORACLE_HOME/bin:$GRID_HOME/bin:$PATH; echo $ORACLE_SID;echo $ORACLE_HOME'
alias oh='cd $ORACLE_HOME;pwd'
alias ss='sqlplus / as sysdba'
```

- **디렉토리 및 권한 설정**

```
[root@jjh2 ~]# mkdir -p /u01/app/oracle/product/19c
[root@jjh2 ~]# mkdir -p /u01/app/grid/19c
[root@jjh2 ~]# mkdir -p /u01/oraInventory
[root@jjh2 ~]# mkdir -p /oraarch
[root@jjh2 ~]# chown -R oracle:dba /u01
[root@jjh2 ~]# chmod -R 775 /u01
[root@jjh2 ~]# chown -R oracle:dba /oraarch
[root@jjh2 ~]# chmod -R 775 /oraarch
[root@jjh2 ~]# chown -R oracle:dba /dev
```

오라클 설치를 위한 디렉토리 생성 및 oracle유저의 /u01 디렉토리에 대한 권한을 설정한다.
